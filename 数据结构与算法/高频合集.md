## 1.模拟实现函数的call、apply、bind方法

// call
Function.prototype.mycall = function(context,...args){
    context = typeof context == 'object' ? context : window
    let key  = Symbol()
    context[key] = this
    let result = context[key](...args)
    delete context[key]
    return result
}

function say(a,b){
    console.log(this.name)
    console.log(a + b)
}

let me = {
    name:'qgq'
}
say.mycall(me,1,2)

// bind
Function.prototype.mybind = function(context,...args){
    context = typeof context == 'object' ? context : window
    return(...args)=>{
        this.call(context,...args)
    }
} 
let li = {
    name:'lili'
}
let myb = say.mybind(li)
myb(1,2)
````
## 2.防抖、节流
````
// 防抖
function debounce(fn,delay){
    let timer = null
    return function(){
        if(timer){
            timer = null
            clearTimeout(timer)
        }
        let context = this
        let args    = arguments
        timer = setTimeout(function(){
           fn.apply(context,args)
        },delay)
    }
}
function fn(){
    console.log('111')
}
document.body.addEventListener('scroll', debounce(fn, 2000));
````

function flatten(){
    let res = []
    return function flat(arr){
        arr.forEach(item =>{
            if(Array.isArray(item)){
                flat(item)
            }else{
                res.push(item)
            }
        })
       return res
    }

function flatten(arr){

## 4.模拟实现promise.all
````
function myall(promises){
    let len = promises.length
    let res = []
    let i   = 0
    return new Promise((resolve,reject)=>{
       for(let item of promises){
           Promise.resolve(item)
           .then((val)=>{
             res[i++] = val
             if(i == len) resolve(res)
           }).catch(e=>{
               reject(e)
           })
       }
    })
}

## 6.二分查找
````
//非递归 
function binary_search(nums,target){
    let left = 0
    let right = nums.length-1
    let mid = Math.floor((right+left)/2)
    while(left < right){
        if(nums[mid] == target){
             return mid
        }else if(nums[mid]>target){
            right = mid-1 
        }else{
            left  = mid+1
        }
        mid = Math.floor((right+left)/2)
    }
}

//递归
function binary_search(nums,target,left,right){
    if(left > right) return false
    let mid = Math.floor((right+left)/2)
    if(nums[mid] == target){
        return mid
    }else if(nums[mid] > target){
        right = mid -1
        return binary_search(nums,target,left,right)
    }else{
        left = mid + 1
        return binary_search(nums,target,left,right)
    }
}

## 7.快排
````
function quickSort(arr){
    let left = []
    let right= []
    if(arr.length<2) return arr
    let len  = arr.length -1
    let mid = arr.splice(Math.floor(len/2),1)[0]
    for(let i=0;i<len;i++){
        if(arr[i]<mid){
            left.push(arr[i])
        }else{
            right.push(arr[i])
        }
    }
    return quickSort(left).concat([mid],quickSort(right))

}
